Tipo(nome, descTipo):
     ---- 
•	CC = {{nome}}

Equipamento(id, eqDesc, tipo): 
	    -- 
•	CP = {{id}}
•	CE = {{tipo REF Tipo(nome)}}
RI 1: Tipo não é nulo; 

Cliente(id, nome, NIF, morada):
	--
•	CP = {{id}}
RI 2: Nome, NIF, morada são obrigatorios, caso não seja "cliente final"
RI 3: Cliente final é o unico que tem nome, nif e morada a nulo

Empregado(numero, nome): 
	  ------
•	CP = {{numero}}

Preço(eqId, valor, duração, validade):
      ----  -----  -------
•	CC = {{eqId, valor, duração}}
•	CE = {{eqId REF Equipamento(id)}}
RI 4: Validade é obrigatoria

Aluguer(eqId, serial, inicio, fim, duração, preço, empregado, cliente):
	----  ------
•	CC = {{eqId, serial}}
•	CE = {{eqId REF Equipamento(id)}}
•	CE = {{empregado REF Empregado(numero)}}
•	CE = {{cliente REF Cliente(id)}}
RI 5: Todos os atributos de aluguer são obrigatorios
RI 6: A data de fim é maior que a data de inicio
RI 7: Fim = inicio + duração
